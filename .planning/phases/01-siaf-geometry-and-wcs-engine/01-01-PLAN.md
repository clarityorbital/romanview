       import siafData from '../data/wfi_siaf.json';

       export interface SIAFDetector {
         id: string;
         v2Ref: number;       // arcseconds
         v3Ref: number;       // arcseconds
         v3IdlYAngle: number; // degrees
         vIdlParity: number;  // +1 or -1
         xSciScale: number;   // arcsec/pixel
         ySciScale: number;   // arcsec/pixel
         xDetSize: number;    // pixels
         yDetSize: number;    // pixels
         corners_v2v3: [number, number][]; // 4 corners in V2V3 arcsec
       }

       export interface SIAFBoresight {
         v2: number;  // arcseconds
         v3: number;  // arcseconds
       }

       export function getAllDetectors(): SIAFDetector[] {
         return Object.entries(siafData.detectors).map(([id, det]) => ({
           id,
           ...(det as Omit<SIAFDetector, 'id'>),
         }));

       export function getDetector(id: string): SIAFDetector | undefined {
         const det = (siafData.detectors as Record<string, Omit<SIAFDetector, 'id'>>)[id];

         return siafData.boresight as SIAFBoresight;

  <action>
    Create `src/lib/wcs.ts` implementing the centralized WCS engine. This is the SINGLE source of truth for ALL coordinate projection math in the application. No other file should implement its own projection.

    The module must export these functions (implementations follow the RESEARCH.md code examples exactly):

    1. `gnomonicForward(ra, dec, ra0, dec0)` -- Sky to tangent plane.
       - Inputs: ra, dec (source, degrees), ra0, dec0 (tangent point, degrees)
       - Returns: `{ xi: number, eta: number }` in radians, or null if behind tangent point
       - Use the Calabretta & Greisen 2002 formulas from RESEARCH.md

    2. `gnomonicInverse(xi, eta, ra0, dec0)` -- Tangent plane to sky.
       - Inputs: xi, eta (tangent plane, radians), ra0, dec0 (tangent point, degrees)
       - Returns: `{ ra: number, dec: number }` in degrees
       - RA normalized to [0, 360)

    3. `rotateByPA(xi, eta, paDeg)` -- Rotate tangent plane coords by position angle.
       - PA measured East of North (standard astronomical convention)
       - Returns: rotated `{ xi, eta }` in same units as input

    4. `v2v3ToSky(v2, v3, boresightV2, boresightV3, targetRa, targetDec, v3paDeg)` -- Convert SIAF V2V3 position to sky RA/Dec.
       - v2, v3, boresightV2, boresightV3 in arcseconds
       - targetRa, targetDec in degrees (where boresight points on sky)
       - v3paDeg in degrees
       - CRITICAL: xi = -dv2 (V2 sign flip) per RESEARCH.md Pattern 3
       - Returns: `{ ra: number, dec: number }` in degrees

    5. `skyToFocalPlane(ra, dec, boresightRa, boresightDec, v3paDeg)` -- Project sky position onto focal plane.
       - Returns: `{ x: number, y: number }` in arcminutes, or null if behind tangent point
       - This replaces the `projectToFocalPlane` function currently in FocalPlaneView.tsx
       - Forward gnomonic projection, then rotate by -V3PA to get focal plane coords
       - Convert from radians to arcminutes
